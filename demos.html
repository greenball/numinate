Earn money

Press start to get rich :3
$('#account').numinate({
	format: 'Your balance is %counter% \$\$',
	from: 500,
	to: 250000,
	runningInterval: 1500,
	stepUnit: 1000,
	autoStart: false,
	autoRemove: false,
	onStop: function(elem) {
		setTimeout(function() {
			$(elem).text('Now pay the taxes...');
		}, 1500);
		
	},
	onComplete: function(elem) {
		setTimeout(function() {
			$(elem).text('Your balance is -500.000 \$\$... Easy come, easy go :P');
		}, 2500);
	},
});
